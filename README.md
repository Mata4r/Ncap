# Windows Installation
- Run CMD as Admin
- git clone https://github.com/Mata4r/Ncap.git
- cd Ncap
- pip install scapy
- python Ncap.py

# Linux Installation
- Run Terminal
- git clone https://github.com/Mata4r/Ncap.git
- cd Ncap
- sudo apt update
- sudo apt install python3 python3-pip -y
- pip3 install scapy
- sudo python3 Ncap.py

# MacOS Installation
- Run Terminal
- git clone https://github.com/Mata4r/Ncap.git
- cd Ncap
- brew install python
- pip3 install scapy
- sudo python3 Ncap.py
